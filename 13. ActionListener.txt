    private void proyectoetiqueta () {

        etiqueta.setText("Ingresa su nombre"); //etiqueta con el texto

        etiqueta.setBounds(140, 50, 200, 30);//posicion de la etiqueta, ancho y altura

        panel01.add(etiqueta); //da el resiltado de la etiqueta en el JPanel
    
    }
